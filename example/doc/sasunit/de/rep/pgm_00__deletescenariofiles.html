Ausführliche Beschreibung
Deletes all appendant files in the log, tst and rep folders.
Called before a scenario is being executed in order to avoid disused files

00004 %MACRO _deletescenariofiles(i_scenariosToRun=
00005 );
00006 %LOCAL l_len l_nobs l_obs l_scnDel l_target l_foldersToDelete l_filesToDelete;
00007 %LET l_target = %_abspath(&g_root, &g_target);
00008  
00009 PROC SQL NOPRINT;
00010 CREATE TABLE scenarioFilesToDelete AS
00011 SELECT scn_id
00012 FROM &i_scenariosToRun AS s
00013 WHERE s.dorun = 1 AND scn_id IN (SELECT scn_id FROM target.scn)
00014 ;
00015 SELECT scn_id into :l_scnDel separated by ','
00016 FROM scenarioFilesToDelete
00017 ;
00018 SELECT count(scn_id) AS cnt into :l_obs
00019 FROM scenarioFilesToDelete
00020 ;
00021 QUIT;
00022  
00023 %IF &l_obs GT 0 %THEN %DO;
00024 %*** Dir for deletion of /rep, /tst and /log files *;
00025 %_dir(i_path=&l_target./rep/, o_out=rep);
00026 %_dir(i_path=&l_target./log/, o_out=log);
00027 %_dir(i_path=&l_target./tst/, o_out=tst);
00028  
00029 %let l_filesToDelete = %sysfunc(pathname(work))/_scenarioFilesToDelete.sas;
00030 DATA _NULL_;
00031 FILE "&l_filesToDelete";
00032 SET scenarioFilesToDelete nobs=numobs_dorun;
00033  
00034 DO i=1 TO numobs_log;
00035 SET log nobs=numobs_log point=i;
00036 IF index(membername, put(scn_id,z3.)) = 1 THEN DO;
00037 PUT '%PUT Delete ' filename ' RC: %_delfile(' filename ');';
00038 END;
00039 END;
00040 DO j=1 TO numobs_rep;
00041 SET rep nobs=numobs_rep point=j;
00042 IF index(membername, put(scn_id,z3.)) = 1 OR
00043 index(membername, catt("_", put(scn_id,z3.))) = 1 OR
00044 index(membername, catt("cas_", put(scn_id,z3.),"_")) = 1 OR
00045 index(membername, catt("tcg_", put(scn_id,z3.))) = 1
00046 THEN DO;
00047 PUT '%PUT Delete ' filename ' RC: %_delfile(' filename ');';
00048 END;
00049 END;
00050 DO k=1 TO numobs_tst;
00051 SET tst nobs=numobs_tst point=k;
00052 IF index(membername, put(scn_id,z3.)) = 1 THEN DO;
00053 PUT '%PUT Delete ' filename ' RC: %_delfile(' filename ');';
00054 END;
00055 END;
00056 RUN;
00057 %INCLUDE "&l_filesToDelete.";
00058 %LET l_rc=%_delfile(&l_filesToDelete.);
00059  
00060 %*** Deletion of /tst folders ***;
00061 %_dir(i_path=&l_target./tst/, i_recursive=1, o_out=tst_folder);
00062  
00063 %LET l_len = %sysfunc(length(&l_target./tst/));
00064  
00065 DATA foldersToDelete;
00066 SET tst_folder;
00067 part = substr(filename, &l_len+1);
00068 pos = index(part, "/");
00069 IF pos > 0 THEN DO;
00070 folder = substr(part, 1, pos-1);
00071 id = input(substr(folder,2,3),3.);
00072 IF id IN (&l_scnDel) THEN DO;
00073 OUTPUT;
00074 END;
00075 END;
00076 KEEP id folder;
00077 RUN;
00078  
00079 %LET l_nobs = %_nobs(foldersToDelete);
00080  
00081 %*** Write and execute cmd file only if table foldersToDelete is not empty ***;
00082 %IF &l_nobs > 0 %THEN %DO;
00083 PROC SORT DATA = foldersToDelete nodupkey;
00084 BY folder;
00085 RUN;
00086  
00087 %let l_foldersToDelete = %sysfunc(pathname(work))/_scenarioFoldersToDelete.cmd;
00088 DATA _NULL_;
00089 FILE "&l_foldersToDelete";
00090 SET foldersToDelete;
00091 PUT "&g_removedir ""&l_target./tst/" folder+(-1)"""&g_endcommand";
00092 RUN;
00093 %_executeCMDFile(&l_foldersToDelete);
00094 %LET l_rc=%_delfile(&l_foldersToDelete);
00095 %END;
00096 %END;
00097 %MEND _deletescenariofiles;
