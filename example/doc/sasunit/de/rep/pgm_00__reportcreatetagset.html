00017 define event row;
00018 finish: /* Output the current row-data gathered within the event 'data' and stored within the associative array 'myRow' */
00019 trigger printScenario / if cmp($myRow["isScenario"], "1");
00020 trigger printTestcase / if cmp($myRow["isScenario"], "0");
00021 UNSET $myRow;
00022 UNSET $failuresUnformatted;
00023 end;

00025 define event data;
00026 /* Store the data of each cell of the current row in the associative array 'myRow' for outputing it in the row.finish-event */
00027 SET $myRow[NAME] VALUE;
00028 SET $failuresUnformatted UNFORMATTEDVALUE / if cmp(NAME,"failures");
