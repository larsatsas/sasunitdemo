00002 %MACRO _abspath (i_root
00003 ,i_path
00004 );
00005  
00006 %IF "&i_path" EQ "" %THEN %RETURN;
00007  
00008 %IF %length(&i_root) %THEN %DO;
00009 %LET i_root = %qsysfunc(translate(&i_root,/,\));
00010 %IF "%substr(&i_root,%length(&i_root),1)" = "/" %THEN
00011 %LET i_root=%substr(&i_root,1,%eval(%length(&i_root)-1));
00012 %END;
00013  
00014 %LET i_path = %qsysfunc(translate(&i_path,/,\));
00015 %IF "%substr(&i_path,1,1)" = "/" OR "%substr(&i_path,2,2)" = ":/" %THEN &i_path;
00016 %ELSE %IF %length(&i_root) %THEN &i_root/&i_path;
00017 %ELSE &i_path;
00018  
00019 %MEND _abspath;
