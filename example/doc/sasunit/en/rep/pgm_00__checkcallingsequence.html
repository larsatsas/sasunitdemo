00001 %MACRO _checkCallingSequence(i_callerType =);
00002  
00003 %global g_inScenario g_inTestCase g_inTestCall;
00004  
00005 %let l_callerType = %lowcase (&i_callerType.);
00006  
00007 %if (&l_callertype.=scenario) %then %do;
00008 %if (&g_inScenario. = 1) %then %do;
00009 %PUT &g_error.: initScenario must not be called twice!;
00010 1
00011 %RETURN;
00012 %end;
00013 %if (&g_inTestCall. = 1) %then %do;
00014 %PUT &g_error.: initScenario must not be called within a test call!;
00015 1
00016 %RETURN;
00017 %end;
00018 %if (&g_inTestcase. = 1) %then %do;
00019 %PUT &g_error.: initScenario must not be called within a test case!;
00020 1
00021 %RETURN;
00022 %end;
00023 0
00024 %RETURN;
00025 %end;
00026 %if (&l_callertype.=assert) %then %do;
00027 %if (&g_inScenario. = 0) %then %do;
00028 %PUT &g_error.: Assert must be called after initScenario!;
00029 1
00030 %RETURN;
00031 %end;
00032 %if (&g_inTestcase. = 0) %then %do;
00033 %PUT &g_error.: Assert must be called after initTestCase!;
00034 1
00035 %RETURN;
00036 %end;
00037 0
00038 %RETURN;
00039 %end;
00040 -1
00041 %MEND _checkcallingSequence;
