00001 %MACRO _copyMacrosToRepSrc (o_pgmdoc_sasunit=);
00002  
00003 %local l_output_path l_rc l_saspgm;
00004  
00005 data work._exa;
00006 set target.exa
00007 %if (&o_pgmdoc_sasunit. ne 1) %then %do;
00008 (where=(exa_auton >= 2 or missing(exa_auton)))
00009 %end;
00010 ;
00011 *** Files not residing in an autocall path will have . as exa_auton ***;
00012 *** Use 99 as an alternative ***;
00013 exa_auton = coalesce (exa_auton, 99);
00014 run;
00015  
00016 proc sort data=work._exa;
00017 by exa_auton exa_pgm;
00018 run;
00019  
00020 %let l_output_path = %_abspath(&g_root., &g_target.)/rep;
00021 %let l_rc = %sysfunc (dcreate (src, &l_output_path.));
00022 %let l_output_path = &l_output_path./src;
00023 %let l_saspgm = %sysfunc(pathname(work))/CopyMacrosToRep.sas;
00024 data _null_;
00025 file "&l_saspgm.";
00026 set work._exa;
00027 length command $32000 NewDir DirRc $300;
00028 Retain NewDir;
00029 by exa_auton;
00030 if (first.exa_auton) then do;
00031 NewDir = "&l_output_path./" !! put (exa_auton, z2.);
00032 DirRc = dcreate (put (exa_auton, z2.), "&l_output_path.");
00033 end;
00034 command = catt ('%_copyfile(', exa_filename, ",", NewDir, "/", exa_pgm, ");");
00035 put command;
00036 run;
00037 %include "&l_saspgm.";
00038  
00039 proc sort data=target.scn (where=(not missing(scn_start))) out=work._scn;
00040 by scn_id;
00041 run;
00042  
00043 %let l_output_path = %_abspath(&g_root., &g_target.)/rep/src;
00044 %let l_rc = %sysfunc (dcreate (scn, &l_output_path.));
00045 %let l_output_path = &l_output_path./scn;
00046 %let l_saspgm = %sysfunc(pathname(work))/CopyMacrosToRep.sas;
00047

00052 scn_abs_path = resolve ('%_abspath(&g_root,' !! trim(scn_path) !! ')');

00054 put command;
00055 run;
00056 %include "&l_saspgm.";
